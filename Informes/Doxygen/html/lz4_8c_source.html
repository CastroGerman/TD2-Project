115 void *memcpy_safe(void* dest, const void* src, int cnt)
116 {
117  int start = (int)dest;
118  int end = start + cnt;
119  if (start <= 0x1FFFFFFF && end >= 0x20000000) // if memory lies over SRAM_L and SRAM_U boundary
120  {
121  volatile uint8_t useless = 0; // force 8 bit copy instead of optimized 32 bit copy
122  volatile uint8_t c;
123  volatile int i = 0;
124  volatile uint8_t* src_p = (uint8_t*)src;
125  volatile uint8_t* dest_p = (uint8_t*)dest;
126  for (i = 0; i < cnt && useless == 0; i++) {
127  c = src_p[i];
128  dest_p[i] = c;
129  }
130  return dest;
131  }
132  else
133  {
134  return memcpy(dest, src, cnt);
135  }
136 }

283 static unsigned LZ4_NbCommonBytes (register size_t val)
284 {
285  if (LZ4_isLittleEndian())
286  {
287  if (LZ4_64bits())
288  {
289 # if defined(_MSC_VER) && defined(_WIN64) && !defined(LZ4_FORCE_SW_BITCOUNT)
290  unsigned long r = 0;
291  _BitScanForward64( &r, (U64)val );
292  return (int)(r>>3);
293 # elif (defined(__clang__) || (LZ4_GCC_VERSION >= 304)) && !defined(LZ4_FORCE_SW_BITCOUNT)
294  return (__builtin_ctzll((U64)val) >> 3);
295 # else
296  static const int DeBruijnBytePos[64] = { 0, 0, 0, 0, 0, 1, 1, 2, 0, 3, 1, 3, 1, 4, 2, 7, 0, 2, 3, 6, 1, 5, 3, 5, 1, 3, 4, 4, 2, 5, 6, 7, 7, 0, 1, 2, 3, 3, 4, 6, 2, 6, 5, 5, 3, 4, 5, 6, 7, 1, 2, 4, 6, 4, 4, 5, 7, 2, 6, 5, 7, 6, 7, 7 };
297  return DeBruijnBytePos[((U64)((val & -(long long)val) * 0x0218A392CDABBD3FULL)) >> 58];
298 # endif
299  }
300  else /* 32 bits */
301  {
302 # if defined(_MSC_VER) && !defined(LZ4_FORCE_SW_BITCOUNT)
303  unsigned long r;
304  _BitScanForward( &r, (U32)val );
305  return (int)(r>>3);
306 # elif (defined(__clang__) || (LZ4_GCC_VERSION >= 304)) && !defined(LZ4_FORCE_SW_BITCOUNT)
307  return (__builtin_ctz((U32)val) >> 3);
308 # else
309  static const int DeBruijnBytePos[32] = { 0, 0, 3, 0, 3, 1, 3, 0, 3, 2, 2, 1, 3, 2, 0, 1, 3, 3, 1, 2, 2, 2, 2, 0, 3, 1, 2, 0, 1, 0, 1, 1 };
310  return DeBruijnBytePos[((U32)((val & -(S32)val) * 0x077CB531U)) >> 27];
311 # endif
312  }
313  }
314  else /* Big Endian CPU */
315  {
316  if (LZ4_64bits())
317  {
318 # if defined(_MSC_VER) && defined(_WIN64) && !defined(LZ4_FORCE_SW_BITCOUNT)
319  unsigned long r = 0;
320  _BitScanReverse64( &r, val );
321  return (unsigned)(r>>3);
322 # elif (defined(__clang__) || (LZ4_GCC_VERSION >= 304)) && !defined(LZ4_FORCE_SW_BITCOUNT)
323  return (__builtin_clzll((U64)val) >> 3);
324 # else
325  unsigned r;
326  if (!(val>>32)) { r=4; } else { r=0; val>>=32; }
327  if (!(val>>16)) { r+=2; val>>=8; } else { val>>=24; }
328  r += (!val);
329  return r;
330 # endif
331  }
332  else /* 32 bits */
333  {
334 # if defined(_MSC_VER) && !defined(LZ4_FORCE_SW_BITCOUNT)
335  unsigned long r = 0;
336  _BitScanReverse( &r, (unsigned long)val );
337  return (unsigned)(r>>3);
338 # elif (defined(__clang__) || (LZ4_GCC_VERSION >= 304)) && !defined(LZ4_FORCE_SW_BITCOUNT)
339  return (__builtin_clz((U32)val) >> 3);
340 # else
341  unsigned r;
342  if (!(val>>16)) { r=2; val>>=8; } else { r=0; val>>=24; }
343  r += (!val);
344  return r;
345 # endif
346  }
347  }
348 }

350 static unsigned LZ4_count(const BYTE* pIn, const BYTE* pMatch, const BYTE* pInLimit)
351 {
352  const BYTE* const pStart = pIn;
353 
354  while (likely(pIn<pInLimit-(STEPSIZE-1)))
355  {
356  size_t diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
357  if (!diff) { pIn+=STEPSIZE; pMatch+=STEPSIZE; continue; }
358  pIn += LZ4_NbCommonBytes(diff);
359  return (unsigned)(pIn - pStart);
360  }
361 
362  if (LZ4_64bits()) if ((pIn<(pInLimit-3)) && (LZ4_read32(pMatch) == LZ4_read32(pIn))) { pIn+=4; pMatch+=4; }
363  if ((pIn<(pInLimit-1)) && (LZ4_read16(pMatch) == LZ4_read16(pIn))) { pIn+=2; pMatch+=2; }
364  if ((pIn<pInLimit) && (*pMatch == *pIn)) pIn++;
365  return (unsigned)(pIn - pStart);
366 }

470 FORCE_INLINE int LZ4_compress_generic(
471  void* const ctx,
472  const char* const source,
473  char* const dest,
474  const int inputSize,
475  const int maxOutputSize,
476  const limitedOutput_directive outputLimited,
477  const tableType_t tableType,
478  const dict_directive dict,
479  const dictIssue_directive dictIssue,
480  const U32 acceleration)
481 {
482  LZ4_stream_t_internal* const dictPtr = (LZ4_stream_t_internal*)ctx;
483 
484  const BYTE* ip = (const BYTE*) source;
485  const BYTE* base;
486  const BYTE* lowLimit;
487  const BYTE* const lowRefLimit = ip - dictPtr->dictSize;
488  const BYTE* const dictionary = dictPtr->dictionary;
489  const BYTE* const dictEnd = dictionary + dictPtr->dictSize;
490  const size_t dictDelta = dictEnd - (const BYTE*)source;
491  const BYTE* anchor = (const BYTE*) source;
492  const BYTE* const iend = ip + inputSize;
493  const BYTE* const mflimit = iend - MFLIMIT;
494  const BYTE* const matchlimit = iend - LASTLITERALS;
495 
496  BYTE* op = (BYTE*) dest;
497  BYTE* const olimit = op + maxOutputSize;
498 
499  U32 forwardH;
500  size_t refDelta=0;
501 
502  /* Init conditions */
503  if ((U32)inputSize > (U32)LZ4_MAX_INPUT_SIZE) return 0; /* Unsupported input size, too large (or negative) */
504  switch(dict)
505  {
506  case noDict:
507  default:
508  base = (const BYTE*)source;
509  lowLimit = (const BYTE*)source;
510  break;
511  case withPrefix64k:
512  base = (const BYTE*)source - dictPtr->currentOffset;
513  lowLimit = (const BYTE*)source - dictPtr->dictSize;
514  break;
515  case usingExtDict:
516  base = (const BYTE*)source - dictPtr->currentOffset;
517  lowLimit = (const BYTE*)source;
518  break;
519  }
520  if ((tableType == byU16) && (inputSize>=LZ4_64Klimit)) return 0; /* Size too large (not within 64K limit) */
521  if (inputSize<LZ4_minLength) goto _last_literals; /* Input too small, no compression (all literals) */
522 
523  /* First Byte */
524  LZ4_putPosition(ip, ctx, tableType, base);
525  ip++; forwardH = LZ4_hashPosition(ip, tableType);
526 
527  /* Main Loop */
528  for ( ; ; )
529  {
530  const BYTE* match;
531  BYTE* token;
532  {
533  const BYTE* forwardIp = ip;
534  unsigned step = 1;
535  unsigned searchMatchNb = acceleration << LZ4_skipTrigger;
536 
537  /* Find a match */
538  do {
539  U32 h = forwardH;
540  ip = forwardIp;
541  forwardIp += step;
542  step = (searchMatchNb++ >> LZ4_skipTrigger);
543 
544  if (unlikely(forwardIp > mflimit)) goto _last_literals;
545 
546  match = LZ4_getPositionOnHash(h, ctx, tableType, base);
547  if (dict==usingExtDict)
548  {
549  if (match<(const BYTE*)source)
550  {
551  refDelta = dictDelta;
552  lowLimit = dictionary;
553  }
554  else
555  {
556  refDelta = 0;
557  lowLimit = (const BYTE*)source;
558  }
559  }
560  forwardH = LZ4_hashPosition(forwardIp, tableType);
561  LZ4_putPositionOnHash(ip, h, ctx, tableType, base);
562 
563  } while ( ((dictIssue==dictSmall) ? (match < lowRefLimit) : 0)
564  || ((tableType==byU16) ? 0 : (match + MAX_DISTANCE < ip))
565  || (LZ4_read32(match+refDelta) != LZ4_read32(ip)) );
566  }
567 
568  /* Catch up */
569  while ((ip>anchor) && (match+refDelta > lowLimit) && (unlikely(ip[-1]==match[refDelta-1]))) { ip--; match--; }
570 
571  {
572  /* Encode Literal length */
573  unsigned litLength = (unsigned)(ip - anchor);
574  token = op++;
575  if ((outputLimited) && (unlikely(op + litLength + (2 + 1 + LASTLITERALS) + (litLength/255) > olimit)))
576  return 0; /* Check output limit */
577  if (litLength>=RUN_MASK)
578  {
579  int len = (int)litLength-RUN_MASK;
580  *token=(RUN_MASK<<ML_BITS);
581  for(; len >= 255 ; len-=255) *op++ = 255;
582  *op++ = (BYTE)len;
583  }
584  else *token = (BYTE)(litLength<<ML_BITS);
585 
586  /* Copy Literals */
587  LZ4_wildCopy(op, anchor, op+litLength);
588  op+=litLength;
589  }
590 
591 _next_match:
592  /* Encode Offset */
593  LZ4_writeLE16(op, (U16)(ip-match)); op+=2;
594 
595  /* Encode MatchLength */
596  {
597  unsigned matchLength;
598 
599  if ((dict==usingExtDict) && (lowLimit==dictionary))
600  {
601  const BYTE* limit;
602  match += refDelta;
603  limit = ip + (dictEnd-match);
604  if (limit > matchlimit) limit = matchlimit;
605  matchLength = LZ4_count(ip+MINMATCH, match+MINMATCH, limit);
606  ip += MINMATCH + matchLength;
607  if (ip==limit)
608  {
609  unsigned more = LZ4_count(ip, (const BYTE*)source, matchlimit);
610  matchLength += more;
611  ip += more;
612  }
613  }
614  else
615  {
616  matchLength = LZ4_count(ip+MINMATCH, match+MINMATCH, matchlimit);
617  ip += MINMATCH + matchLength;
618  }
619 
620  if ((outputLimited) && (unlikely(op + (1 + LASTLITERALS) + (matchLength>>8) > olimit)))
621  return 0; /* Check output limit */
622  if (matchLength>=ML_MASK)
623  {
624  *token += ML_MASK;
625  matchLength -= ML_MASK;
626  for (; matchLength >= 510 ; matchLength-=510) { *op++ = 255; *op++ = 255; }
627  if (matchLength >= 255) { matchLength-=255; *op++ = 255; }
628  *op++ = (BYTE)matchLength;
629  }
630  else *token += (BYTE)(matchLength);
631  }

634 
635  /* Test end of chunk */
636  if (ip > mflimit) break;

639  LZ4_putPosition(ip-2, ctx, tableType, base);
640 
641  /* Test next position */
642  match = LZ4_getPosition(ip, ctx, tableType, base);
643  if (dict==usingExtDict)
644  {
645  if (match<(const BYTE*)source)
646  {
647  refDelta = dictDelta;
648  lowLimit = dictionary;
649  }
650  else
651  {
652  refDelta = 0;
653  lowLimit = (const BYTE*)source;
654  }
655  }
656  LZ4_putPosition(ip, ctx, tableType, base);
657  if ( ((dictIssue==dictSmall) ? (match>=lowRefLimit) : 1)
658  && (match+MAX_DISTANCE>=ip)
659  && (LZ4_read32(match+refDelta)==LZ4_read32(ip)) )
660  { token=op++; *token=0; goto _next_match; }
661 
662  /* Prepare next loop */
663  forwardH = LZ4_hashPosition(++ip, tableType);
664  }

666 _last_literals:

668  {
669  const size_t lastRun = (size_t)(iend - anchor);
670  if ((outputLimited) && ((op - (BYTE*)dest) + lastRun + 1 + ((lastRun+255-RUN_MASK)/255) > (U32)maxOutputSize))

673  {
674  size_t accumulator = lastRun - RUN_MASK;
675  *op++ = RUN_MASK << ML_BITS;
676  for(; accumulator >= 255 ; accumulator-=255) *op++ = 255;
677  *op++ = (BYTE) accumulator;
678  }
679  else
680  {
681  *op++ = (BYTE)(lastRun<<ML_BITS);
682  }
683  memcpy(op, anchor, lastRun);
684  op += lastRun;
685  }
686 
687  /* End */
688  return (int) (((char*)op)-dest);

757 static int LZ4_compress_destSize_generic(
758  void* const ctx,
759  const char* const src,
760  char* const dst,
761  int* const srcSizePtr,
762  const int targetDstSize,
763  const tableType_t tableType)
764 {
765  const BYTE* ip = (const BYTE*) src;
766  const BYTE* base = (const BYTE*) src;
767  const BYTE* lowLimit = (const BYTE*) src;
768  const BYTE* anchor = ip;
769  const BYTE* const iend = ip + *srcSizePtr;
770  const BYTE* const mflimit = iend - MFLIMIT;
771  const BYTE* const matchlimit = iend - LASTLITERALS;
772 
773  BYTE* op = (BYTE*) dst;
774  BYTE* const oend = op + targetDstSize;
775  BYTE* const oMaxLit = op + targetDstSize - 2 /* offset */ - 8 /* because 8+MINMATCH==MFLIMIT */ - 1 /* token */;
776  BYTE* const oMaxMatch = op + targetDstSize - (LASTLITERALS + 1 /* token */);
777  BYTE* const oMaxSeq = oMaxLit - 1 /* token */;
778 
779  U32 forwardH;
780 
781 
782  /* Init conditions */
783  if (targetDstSize < 1) return 0; /* Impossible to store anything */
784  if ((U32)*srcSizePtr > (U32)LZ4_MAX_INPUT_SIZE) return 0; /* Unsupported input size, too large (or negative) */
785  if ((tableType == byU16) && (*srcSizePtr>=LZ4_64Klimit)) return 0; /* Size too large (not within 64K limit) */
786  if (*srcSizePtr<LZ4_minLength) goto _last_literals; /* Input too small, no compression (all literals) */
787 
788  /* First Byte */
789  *srcSizePtr = 0;
790  LZ4_putPosition(ip, ctx, tableType, base);
791  ip++; forwardH = LZ4_hashPosition(ip, tableType);
792 
793  /* Main Loop */
794  for ( ; ; )
795  {
796  const BYTE* match;
797  BYTE* token;
798  {
799  const BYTE* forwardIp = ip;
800  unsigned step = 1;
801  unsigned searchMatchNb = 1 << LZ4_skipTrigger;
802 
803  /* Find a match */
804  do {
805  U32 h = forwardH;
806  ip = forwardIp;
807  forwardIp += step;
808  step = (searchMatchNb++ >> LZ4_skipTrigger);
809 
810  if (unlikely(forwardIp > mflimit))
811  goto _last_literals;
812 
813  match = LZ4_getPositionOnHash(h, ctx, tableType, base);
814  forwardH = LZ4_hashPosition(forwardIp, tableType);
815  LZ4_putPositionOnHash(ip, h, ctx, tableType, base);
816 
817  } while ( ((tableType==byU16) ? 0 : (match + MAX_DISTANCE < ip))
818  || (LZ4_read32(match) != LZ4_read32(ip)) );
819  }
820 
821  /* Catch up */
822  while ((ip>anchor) && (match > lowLimit) && (unlikely(ip[-1]==match[-1]))) { ip--; match--; }
823 
824  {
825  /* Encode Literal length */
826  unsigned litLength = (unsigned)(ip - anchor);
827  token = op++;
828  if (op + ((litLength+240)/255) + litLength > oMaxLit)
829  {
830  /* Not enough space for a last match */
831  op--;
832  goto _last_literals;
833  }
834  if (litLength>=RUN_MASK)
835  {
836  unsigned len = litLength - RUN_MASK;
837  *token=(RUN_MASK<<ML_BITS);
838  for(; len >= 255 ; len-=255) *op++ = 255;
839  *op++ = (BYTE)len;
840  }
841  else *token = (BYTE)(litLength<<ML_BITS);
842 
843  /* Copy Literals */
844  LZ4_wildCopy(op, anchor, op+litLength);
845  op += litLength;
846  }
847 
848 _next_match:
849  /* Encode Offset */
850  LZ4_writeLE16(op, (U16)(ip-match)); op+=2;
851 
852  /* Encode MatchLength */
853  {
854  size_t matchLength;
855 
856  matchLength = LZ4_count(ip+MINMATCH, match+MINMATCH, matchlimit);
857 
858  if (op + ((matchLength+240)/255) > oMaxMatch)
859  {
860  /* Match description too long : reduce it */
861  matchLength = (15-1) + (oMaxMatch-op) * 255;
862  }
863  //printf("offset %5i, matchLength%5i \n", (int)(ip-match), matchLength + MINMATCH);
864  ip += MINMATCH + matchLength;
865 
866  if (matchLength>=ML_MASK)
867  {
868  *token += ML_MASK;
869  matchLength -= ML_MASK;
870  while (matchLength >= 255) { matchLength-=255; *op++ = 255; }
871  *op++ = (BYTE)matchLength;
872  }
873  else *token += (BYTE)(matchLength);
874  }
875 
876  anchor = ip;
877 
878  /* Test end of block */
879  if (ip > mflimit) break;
880  if (op > oMaxSeq) break;
881 
882  /* Fill table */
883  LZ4_putPosition(ip-2, ctx, tableType, base);
884 
885  /* Test next position */
886  match = LZ4_getPosition(ip, ctx, tableType, base);
887  LZ4_putPosition(ip, ctx, tableType, base);
888  if ( (match+MAX_DISTANCE>=ip)
889  && (LZ4_read32(match)==LZ4_read32(ip)) )
890  { token=op++; *token=0; goto _next_match; }
891 
892  /* Prepare next loop */
893  forwardH = LZ4_hashPosition(++ip, tableType);
894  }
895 
896 _last_literals:
897  /* Encode Last Literals */
898  {
899  size_t lastRunSize = (size_t)(iend - anchor);
900  if (op + 1 /* token */ + ((lastRunSize+240)/255) /* litLength */ + lastRunSize /* literals */ > oend)
901  {
902  /* adapt lastRunSize to fill 'dst' */
903  lastRunSize = (oend-op) - 1;
904  lastRunSize -= (lastRunSize+240)/255;
905  }
906  ip = anchor + lastRunSize;
907 
908  if (lastRunSize >= RUN_MASK)
909  {
910  size_t accumulator = lastRunSize - RUN_MASK;
911  *op++ = RUN_MASK << ML_BITS;
912  for(; accumulator >= 255 ; accumulator-=255) *op++ = 255;
913  *op++ = (BYTE) accumulator;
914  }
915  else
916  {
917  *op++ = (BYTE)(lastRunSize<<ML_BITS);
918  }
919  memcpy(op, anchor, lastRunSize);
920  op += lastRunSize;
921  }
922 
923  /* End */
924  *srcSizePtr = (int) (((const char*)ip)-src);
925  return (int) (((char*)op)-dst);
926 }

1137 * Decompression functions
1138 *******************************/
1139 /*
1140  * This generic decompression function cover all use cases.
1141  * It shall be instantiated several times, using different sets of directives
1142  * Note that it is essential this generic function is really inlined,
1143  * in order to remove useless branches during compilation optimization.
1144  */
1145 FORCE_INLINE int LZ4_decompress_generic(
1146  const char* const source,
1147  char* const dest,
1148  int inputSize,
1149  int outputSize, /* If endOnInput==endOnInputSize, this value is the max size of Output Buffer. */
1150 
1151  int endOnInput, /* endOnOutputSize, endOnInputSize */
1152  int partialDecoding, /* full, partial */
1153  int targetOutputSize, /* only used if partialDecoding==partial */
1154  int dict, /* noDict, withPrefix64k, usingExtDict */
1155  const BYTE* const lowPrefix, /* == dest if dict == noDict */
1156  const BYTE* const dictStart, /* only if dict==usingExtDict */
1157  const size_t dictSize /* note : = 0 if noDict */
1158  )
1159 {
1160  /* Local Variables */
1161  const BYTE* ip = (const BYTE*) source;
1162  const BYTE* const iend = ip + inputSize;
1163 
1164  BYTE* op = (BYTE*) dest;
1165  BYTE* const oend = op + outputSize;
1166  BYTE* cpy;
1167  BYTE* oexit = op + targetOutputSize;
1168  const BYTE* const lowLimit = lowPrefix - dictSize;
1169 
1170  const BYTE* const dictEnd = (const BYTE*)dictStart + dictSize;
1171  const size_t dec32table[] = {4, 1, 2, 1, 4, 4, 4, 4};
1172  const size_t dec64table[] = {0, 0, 0, (size_t)-1, 0, 1, 2, 3};
1173 
1174  const int safeDecode = (endOnInput==endOnInputSize);
1175  const int checkOffset = ((safeDecode) && (dictSize < (int)(64 KB)));
1176 
1177 
1178  /* Special cases */
1179  if ((partialDecoding) && (oexit> oend-MFLIMIT)) oexit = oend-MFLIMIT; /* targetOutputSize too high => decode everything */
1180  if ((endOnInput) && (unlikely(outputSize==0))) return ((inputSize==1) && (*ip==0)) ? 0 : -1; /* Empty output buffer */
1181  if ((!endOnInput) && (unlikely(outputSize==0))) return (*ip==0?1:-1);
1182 
1183 
1184  /* Main Loop */
1185  while (1)
1186  {
1187  unsigned token;
1188  size_t length;
1189  const BYTE* match;
1190 
1191  /* get literal length */
1192  token = *ip++;
1193  if ((length=(token>>ML_BITS)) == RUN_MASK)
1194  {
1195  unsigned s;
1196  do
1197  {
1198  s = *ip++;
1199  length += s;
1200  }
1201  while (likely((endOnInput)?ip<iend-RUN_MASK:1) && (s==255));
1202  if ((safeDecode) && unlikely((size_t)(op+length)<(size_t)(op))) goto _output_error; /* overflow detection */
1203  if ((safeDecode) && unlikely((size_t)(ip+length)<(size_t)(ip))) goto _output_error; /* overflow detection */
1204  }
1205 
1206  /* copy literals */
1207  cpy = op+length;
1208  if (((endOnInput) && ((cpy>(partialDecoding?oexit:oend-MFLIMIT)) || (ip+length>iend-(2+1+LASTLITERALS))) )
1209  || ((!endOnInput) && (cpy>oend-COPYLENGTH)))
1210  {
1211  if (partialDecoding)
1212  {
1213  if (cpy > oend) goto _output_error; /* Error : write attempt beyond end of output buffer */
