83 /* FreeRTOS stack overflow hook */
84 __WEAK__ void vApplicationStackOverflowHook(xTaskHandle pxTask, signed char *pcTaskName)
85 {
86  (void) pxTask;
87  (void) pcTaskName;
88 
89  DEBUGOUT("DIE:ERROR:FreeRTOS: Stack overflow in task %s\r\n", pcTaskName);
90  /* Run time stack overflow checking is performed if
91  configCHECK_FOR_STACK_OVERFLOW is defined to 1 or 2. This hook
92  function is called if a stack overflow is detected. */
93  taskDISABLE_INTERRUPTS();
94  for (;; ) {}
95 }
